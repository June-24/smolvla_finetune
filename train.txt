    Step 2 — Merge chunks

    python - << 'EOF'
    import pyarrow.parquet as pq, pyarrow as pa, json
    from pathlib import Path

    chunks = ["data/chunk_0to9","data/chunk_10to19","data/chunk_20to29","data/chunk_30to39"]
    tables = [pq.read_table(f"{c}/data.parquet") for c in chunks if (Path(c)/"data.parquet").exists()]
    merged = pa.concat_tables(tables)
    Path("data/libero_full").mkdir(exist_ok=True)
    pq.write_table(merged, "data/libero_full/data.parquet")

    task_names = {}
    for c in chunks:
        tf = Path(c)/"task_names.json"
        if tf.exists():
            task_names.update(json.load(open(tf)))
    json.dump(task_names, open("data/libero_full/task_names.json","w"), indent=2)
    print(f"Merged {len(merged)} rows, {len(task_names)} tasks")
    EOF
